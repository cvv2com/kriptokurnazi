from icrawler.builtin import GoogleImageCrawler, BingImageCrawler, BaiduImageCrawler
from icrawler import ImageDownloader
import os
import hashlib
import mimetypes
import time
from urllib.parse import urlparse


# Özel downloader: dosya uzantısını korur, hash ile adlandırır
class FlatImageDownloader(ImageDownloader):
def get_filename(self, task, default_ext):
parsed = urlparse(task['file_url'])
ext = os.path.splitext(parsed.path)[1]
if not ext:
ext = mimetypes.guess_extension(task['headers'].get('Content-Type', '').split(';')[0])
if not ext:
ext = default_ext
url_hash = hashlib.md5(task['file_url'].encode('utf-8')).hexdigest()
return f"{url_hash}{ext}"


def download(self, task, default_ext, timeout=5, **kwargs):
return super().download(task, default_ext, timeout=timeout, **kwargs)

# --- Özel downloader: hash'li adlandırma, tekrar kontrolü, CSV log, OCR ---
class FlatImageDownloader(ImageDownloader):
def __init__(self, *args, **kwargs):
super().__init__(*args, **kwargs)
self.keyword = "unknown"
self.start_time = None
self.root_dir = self.storage.root_dir
os.makedirs(self.root_dir, exist_ok=True) # Klasör garantisi
self.log_path = os.path.join(self.root_dir, "download_log.csv")
if not os.path.exists(self.log_path):
with open(self.log_path, "w", newline="", encoding="utf-8") as csvfile:
writer = csv.writer(csvfile)
writer.writerow(["timestamp", "status", "filename", "keyword", "url", "ocr_text"])

def get_filename(self, task, default_ext):
parsed = urlparse(task['file_url'])
ext = os.path.splitext(parsed.path)[1]
if not ext or len(ext) > 10:
content_type = task.get('headers', {}).get('Content-Type', '')
ext = mimetypes.guess_extension(content_type.split(';')[0])
if not ext:
ext = default_ext
url_hash = hashlib.md5(task['file_url'].encode('utf-8')).hexdigest()
return f"{url_hash}{ext}"

def download(self, task, default_ext, timeout=5, **kwargs):
timestamp = datetime.now()
if self.start_time and timestamp < self.start_time:
return False

filename = self.get_filename(task, default_ext)
file_path = os.path.join(self.root_dir, filename)

if os.path.exists(file_path):
self.logger.info(f"skip downloading file {filename} (already exists)")
print(f"[⏭] Skipped (exists): {filename}")
self.log_download(task['file_url'], filename, skipped=True)
return True

result = super().download(task, default_ext, timeout=timeout, **kwargs)
if result:
print(f"[✓] Downloaded: {filename} ← {task['file_url']}")
self.log_download(task['file_url'], filename)
return result

def extract_text(self, image_path):
try:
image = Image.open(image_path)
text = pytesseract.image_to_string(image)
return text.strip().replace('\n', ' ')
except Exception as e:
return f"[OCR error: {e}]"

def log_download(self, url, filename, skipped=False):
status = "SKIPPED" if skipped else "DOWNLOADED"
timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
file_path = os.path.join(self.root_dir, filename)
ocr_text = ""
if not skipped and os.path.exists(file_path):
ocr_text = self.extract_text(file_path)
with open(self.log_path, "a", newline="", encoding="utf-8") as csvfile:
writer = csv.writer(csvfile)
writer.writerow([timestamp, status, filename, self.keyword, url, ocr_text])
